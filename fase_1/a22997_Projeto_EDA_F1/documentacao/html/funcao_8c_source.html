27ANT* LerLista(const char* nomeFicheiro, char matriz[MAX_LINHAS][MAX_COLUNAS], int* linhas, int* colunas) {
28 FILE* ficheiro = fopen(nomeFicheiro, "r");
29 if (ficheiro == NULL) {
30 perror("Erro ao abrir o ficheiro");
31 return NULL;
32 }
33
34 ANT* lista = NULL;
35 char linha[MAX_COLUNAS];
36 int y = 0;
37 int maxColunas = 0;
38
39 while (fgets(linha, sizeof(linha), ficheiro)) {
40 int tamLinha = strlen(linha);
41 if (linha[tamLinha - 1] == '\n') {
42 linha[tamLinha - 1] = '\0'; // Remover quebra de linha
43 tamLinha--;
44 }
45
46 if (tamLinha > maxColunas) {
47 maxColunas = tamLinha;
48 }
49
50 for (int x = 0; x < tamLinha; x++) {
51 matriz[y][x] = linha[x]; // Armazena na matriz
52 if (linha[x] != '.') {
53 inserirAntena(&lista, linha[x], x, y);
54 }
55 }
56
57 // Preencher com '.' os espaços não usados
58 for (int x = tamLinha; x < MAX_COLUNAS; x++) {
59 matriz[y][x] = '.';
60 }
61 y++;
62 }
63
64 *linhas = y;
65 *colunas = maxColunas;
66 fclose(ficheiro);
67 return lista;
68}
